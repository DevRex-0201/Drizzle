  const poolConnection = mysql.createPool({
    host: 'localhost',
    user: 'root',
    database: 'test',
  });

  const db = drizzle(poolConnection);

  // --- Or you can use mysql2 client ---
  const connection = await mysql.createConnection({
    host: 'localhost',
    user: 'root',
    database: 'test',
  });

  const db = drizzle(connection);

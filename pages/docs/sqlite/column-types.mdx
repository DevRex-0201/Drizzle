### The list of all column types. 

You can also create custom types - [see here](../custom-types).

#### integer

<Section>
```typescript
const table = sqliteTable('table', {
  id: integer('id')
});

// you can customize integer mode be either number or timestamp
integer('id', { mode: 'number' })
integer('id', { mode: 'timestamp' }) // Date
```
```sql
CREATE TABLE table (
	`id` integer
);
```
</Section>

<Section>
```typescript
// to make integer primary key auto increment
integer('id', { mode: 'number' }).primaryKey({ autoIncrement: true })
```
```sql
CREATE TABLE table (
	`id` integer PRIMARY KEY AUTOINCREMENT NOT NULL
);
```
</Section>



#### real

#### text 

<Section>
```typescript
const table = sqliteTable('table', {
  text: text('text')
});

text('text')

text<'union' | 'string' | 'type'>('union')
```

```sql
CREATE TABLE table (
	`text` text
);
```
</Section>

#### blob

<Section>
```typescript
const table = sqliteTable('table', {
  blob: blob('blob')
});

blob('blob')
blob('blob', { mode: 'buffer' })
blob('blob', { mode: 'json' })
blob<{ foo: string }>('blob', { mode: 'json' })
```

```sql
CREATE TABLE table (
  `blob` blob
);
```
</Section>

### Columns attributes

#### Primary key

<Section>
```typescript
const table = sqliteTable('table', { id: integer('id').primaryKey({ autoIncrement: true }) });
```

```sql
CREATE TABLE table (
	`id` integer PRIMARY KEY AUTOINCREMENT NOT NULL
);
```
</Section>

#### Not null

<Section>
```typescript
const table = sqliteTable('table', { numInt: integer('numInt').notNull() });
```

```sql
CREATE TABLE table (
	`numInt` integer NOT NULL
);
```
</Section>

#### Default value

<Section>
```typescript
const table = sqliteTable('table', { numInt: integer('numInt').default(123) });
```

```sql
CREATE TABLE table (
	`numInt` integer DEFAULT 123
);
```
</Section>
